  def self.from_s(str)
    splitted = str.split("::")
    name = splitted[0]
    num_nodes = Integer(splitted[1])
    start_time = Time._load(Base64.decode64(splitted[2])) 
    end_time = Time._load(Base64.decode64(splitted[3]))
    return NeptuneJobData.new(name, num_nodes, start_time, end_time)
  end
